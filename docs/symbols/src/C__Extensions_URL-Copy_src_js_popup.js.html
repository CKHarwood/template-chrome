1 /**
  2  * <p>Responsible for the popup page.</p>
  3  * @author <a href="http://github.com/neocotic">Alasdair Mercer</a>
  4  * @since 0.0.2.1
  5  * @namespace
  6  */
  7 var popup = {
  8 
  9     /**
 10      * <p>Initializes the popup page.</p>
 11      * <p>This involves inserting and configuring the UI elements as well as the
 12      * insertion of localized Strings.</p>
 13      * @requires jQuery
 14      */
 15     init: function () {
 16         var bg = chrome.extension.getBackgroundPage();
 17         // Inserts list items created for all enabled features
 18         for (var i = 0; i < bg.clipboard.features.length; i++) {
 19             var feature = bg.clipboard.features[i];
 20             if (feature.isEnabled()) {
 21                 $('#item ul').append(feature.html);
 22             }
 23         }
 24         // Binds request to click event
 25         $('#item li').click(function () {
 26             chrome.extension.sendRequest({'feature': $(this).attr('name')});
 27         });
 28         // Inserts localized Strings
 29         utils.i18nReplace('#copyAnchorText', 'copy_anchor');
 30         utils.i18nReplace('#copyBBCodeText', 'copy_bbcode');
 31         utils.i18nReplace('#copyEncodedUrlText', 'copy_encoded');
 32         utils.i18nReplace('#copyShortUrlText', 'copy_short');
 33         utils.i18nReplace('#copyUrlText', 'copy_url');
 34         utils.i18nReplace('#ieTabText', 'ie_tab');
 35         utils.i18nReplace('#shortening', 'shortening');
 36         // Shows IE Tab indicator if extension is detected for the selected tab
 37         chrome.tabs.getSelected(null, function (tab) {
 38             if (bg.ietab.isActive(tab)) {
 39                 $('#ieTabItem').show();
 40             }
 41         });
 42         // Shows all keyboard shortcuts if user enabled option
 43         if (utils.get('settingShortcut')) {
 44             $('.shortcut').show();
 45             // TODO: Change to improve L&F on Mac
 46             $('body').css('min-width', '190px');
 47         }
 48         // Ensures consistent widths keeping the popup tidy
 49         popup.resizeText();
 50     },
 51 
 52     /**
 53      * <p>Calculates the widest text <code><div/></code> in the popup and
 54      * assigns that width to all others.</p>
 55      * @requires jQuery
 56      * @private
 57      */
 58     resizeText: function () {
 59         var width = 0;
 60         $('li .text').each(function () {
 61             var scrollWidth = this.scrollWidth;
 62             if (scrollWidth > width) {
 63                 width = scrollWidth;
 64             }
 65         });
 66         $('li .text').css('width', width + 'px');
 67     }
 68 
 69 };
