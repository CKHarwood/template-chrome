1 /**
  2  * <p>Provides utility functions used throughout the extension.</p>
  3  * @author <a href="http://github.com/neocotic">Alasdair Mercer</a>
  4  * @since 0.0.2.1
  5  * @namespace
  6  */
  7 var utils = {
  8 
  9     /**
 10      * <p>Retrieves the value associated with the specified key from
 11      * localStorage.</p>
 12      * <p>If the value is found it is parsed as JSON before being returned;
 13      * otherwise undefined is returned.</p>
 14      * @param {String} key The key of the value to be returned.
 15      * @returns The parsed value associated with the specified key or undefined
 16      * if none exists.
 17      * @see JSON.parse
 18      */
 19     get: function (key) {
 20         var value = localStorage[key];
 21         if (typeof(value) !== 'undefined') {
 22             return JSON.parse(value);
 23         }
 24         return value;
 25     },
 26 
 27     /**
 28      * <p>Replaces the inner HTML of the selected element(s) with the localized
 29      * String looked up from Chrome.</p>
 30      * @param {String} selector A String containing a jQuery selector expression
 31      * to select the element(s) to be modified.
 32      * @param {String} name The name of the localized message to be retrieved.
 33      * @param {String|String[]} [sub] The String(s) to substituted in the
 34      * returned message (where applicable).
 35      * @returns {jQuery} The modified element(s) wrapped in jQuery.
 36      * @requires jQuery
 37      */
 38     i18nReplace: function (selector, name, sub) {
 39         if (typeof(sub) !== 'undefined') {
 40             return $(selector).html(chrome.i18n.getMessage(name, sub));
 41         }
 42         return $(selector).html(chrome.i18n.getMessage(name));
 43     },
 44 
 45     /**
 46      * <p>Initializes the value of the specified key in localStorage.</p>
 47      * <p>If the value is currently undefined the specified default value will
 48      * be assigned to it; otherwise it is reassigned to itself.</p>
 49      * @param {String} key The key whose value is to be initialized.
 50      * @param defaultValue The value to be assigned to the specified key if it
 51      * is currently undefined.
 52      * @returns The previous value associated with the specified key. This will
 53      * be undefined if not previously initialized.
 54      * @see utils.set
 55      */
 56     init: function (key, defaultValue) {
 57         var value = utils.get(key);
 58         if (typeof(value) === 'undefined') {
 59             value = defaultValue;
 60         }
 61         return utils.set(key, value);
 62     },
 63 
 64     /**
 65      * <p>Sets the value of the specified key in localStorage.</p>
 66      * <p>If the specified value is undefined it is assigned directly to the
 67      * key; otherwise it is transformed to a JSON String.</p>
 68      * @param {String} key The key whose value is to be set.
 69      * @param value The value to be assigned to the specified key.
 70      * @returns The previous value associated with the specified key or
 71      * undefined if there was none.
 72      * @see JSON.stringify
 73      */
 74     set: function (key, value) {
 75         var oldValue = utils.get(key);
 76         if (typeof(value) !== 'undefined') {
 77             localStorage[key] = JSON.stringify(value);
 78         } else {
 79             localStorage[key] = value;
 80         }
 81         return oldValue;
 82     }
 83 
 84 };
