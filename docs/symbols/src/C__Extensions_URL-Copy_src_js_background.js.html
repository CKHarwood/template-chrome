 39     /**

 56      * @param {Tab} tab The tab whose information is to be used to generate the

 59      */
 60     copyAnchor: function (tab) {
 61         var data = {href: tab.url};
 62         data.text = tab.title || data.href;

 67         data.text = helper.replaceEntities(data.text);

166      * <p>Returns the information for the active URL Shortener service.</p>

170         // Attempts to lookup enabled URL Shortener service
171         for (var p in shortener) {
172             if (shortener.hasOwnProperty(p)) {
173                 var op = shortener[p];
174                 if (op.isEnabled()) {
175                     return op;
176                 }
177             }

179         // Returns google service by default
180         return shortener.google;

323     /**
324      * <p>Helper function for the internal/external request listeners.</p>
325      * <p>This function will serve the copy request.</p>
326      * @param {Object} request The request sent by the calling script.
327      * @param {String} request.feature The copy request feature to be served.
328      * @param {MessageSender} [sender] An object containing information about the
329      * script context that sent a message or request.
330      * @param {function} [sendResponse] Function to call when you have a
331      * response. The argument should be any JSON-ifiable object, or undefined if
332      * there is no response.
333      * @requires jQuery
334      * @private
335      */
336     onRequestHelper: function (request, sender, sendResponse) {
337         chrome.tabs.getSelected(null, function (tab) {
338             var handler = (ietab.isActive(tab)) ? ietab : clipboard;
339             var popup = chrome.extension.getViews({type: 'popup'})[0];
340             switch (request.feature) {
341                 case feature.url.name:
342                     clipboard.feature = feature.url.name;
343                     handler.copyUrl(tab);
344                     popup.close();
345                     break;
346                 case feature.short.name:
347                     $('#loadDiv', popup.document).show();
348                     $('#item', popup.document).hide();
349                     clipboard.feature = feature.short.name;
350                     handler.copyShortUrl(tab);
351                     break;
352                 case feature.anchor.name:
353                     clipboard.feature = feature.anchor.name;
354                     handler.copyAnchor(tab);
355                     popup.close();
356                     break;
357                 case feature.bbcode.name:
358                     clipboard.feature = feature.bbcode.name;
359                     handler.copyBBCode(tab);
360                     popup.close();
361                     break;
362                 case feature.encoded.name:
363                     clipboard.feature = feature.encoded.name;
364                     handler.copyEncoded(tab);
365                     popup.close();
366                     break;
367             }
368         });
369     },

559 /**

567     /**
568      * <p>Prepends slashes to any escape characters in the string provided.</p>
569      * @param {String} str The string to be parsed.
570      * @returns {String} The parsed string.
571      */
572     addSlashes: function (str) {
573         return str
574             .replace('\\', '\\\\')
575             .replace('"', '\\"')
576             .replace('\'', '\\\'');
577     },
578 
579     /**
580      * <p>Calls the active URL Shortener service with the URL provided in order
581      * to obtain a corresponding short URL.</p>
582      * <p>This function also handles the result of the call by either copying
583      * the returned URL to the clipboard or showing a failure message (if
584      * notifications are enabled by the user) as well as ensuring the popup page
585      * is closed.</p>
586      * @param {String} url The URL to be shortened and added to the clipboard.
587      * @see shortener
588      */
589     callUrlShortener: function (url) {
590         var req = new XMLHttpRequest();
591         var service = clipboard.getUrlShortener();
592         req.open(service.method, service.url, true);
593         req.onreadystatechange = function () {
594             if (req.readyState === 4) {
595                 var output = service.output(req.responseText);
596                 if (output) {
597                     clipboard.copy(output);
598                 } else {
599                     clipboard.status = false;
600                     clipboard.showNotification();
601                 }
602                 var popup = chrome.extension.getViews({type: 'popup'})[0];
603                 popup.close();
604             }
605         };
606         req.setRequestHeader('Content-Type', service.contentType);
607         req.send(service.input(url));
608     },
609 
610     /**
611      * <p>Counts the number of properties that are associated with the object
612      * provided.</p>
613      * <p>This function only counts properties that actually belong to the
614      * object and does not count properties added via prototype.</p>
615      * @param {Object} obj The object to be used.
616      * @return {Integer} The number of properties belonging to the object
617      * provided.
618      */
619     countProperties: function (obj) {
620         var count = 0;
621         for (var p in obj) {
622             if (obj.hasOwnProperty(p)) {
623                 count++;
624             }
625         }
626         return count;
627     },

629     /**
630      * <p>Attempts to create formatted HTML for an anchor tag using the data
631      * provided.</p>
632      * @param {Object} data The information to be used to create the anchor.
633      * @param {String} data.href The URL to be associated with the anchor.
634      * @param {String} data.text The text contents of the anchor.
635      * @param {String} [data.title] The title of the anchor.
636      * @returns {String} The formatted HTML for the generated anchor tag.
637      * @requires jQuery
638      */
639     createAnchor: function (data) {
640         if (!data.target && utils.get('settingTargetAttr')) {
641             data.target = '_blank';
642         }
643         return $('<div/>').append($('<a/>', data)).html();
644     },

646     /**
647      * <p>Attempts to create formatted BBCode for a <code>url</code> tag using
648      * the data provided.</p>
649      * @param {Object} data The information to be used to create the BBCode.
650      * @param {String} [data.text] The text contents of the BBCode.
651      * @param {String} data.url The URL to be associated with the BBCode.
652      * @returns {String} The formatted BBCode for the generated <code>url</code>
653      * tag.
654      */
655     createBBCode: function (data) {
656         if (data.text) {
657             return '[url=' + data.url + ']' + data.text + '[/url]';
658         } else {
659             return '[url]' + data.url + '[/url]';
660         }
661     },

663     /**
664      * <p>Creates a <code><li/></code> representing the feature provided.
665      * This is to be inserted in to the <code><ul/></code> in the popup
666      * page but is created here to optimize display times for the popup.</p>
667      * @param {feature} feature The information of the feature to be used.
668      * @returns {String} The formatted HTML for the specified feature.
669      * @requires jQuery
670      * @private
671      */
672     createFeatureHtml: function (feature) {
673         var item = $('<li/>', {
674             'id': feature.id + 'Item',
675             'name': feature.name
676         });
677         var menu = $('<div/>', {
678             'class': 'menu',
679             'id': feature.id
680         });
681         menu.append($('<span/>', {
682             'class': 'text',
683             'id': feature.id + 'Text'
684         }));
685         var shortcut = feature.getShortcut();
686         if (clipboard.isThisPlatform('mac')) {
687             shortcut = feature.getMacShortcut();
688         }
689         menu.append($('<span/>', {
690             'class': 'shortcut',
691             'id': feature.id + 'Shortcut',
692             'text': shortcut
693         }));
694         item.append(menu);
695         return $('<div/>').append(item).html();
696     },

725      * @param {String} str The string to be parsed.
726      * @returns {String} The parsed string.
727      */
728     replaceEntities: function (str) {
729         return str
730             .replace('"', '"')
731             .replace('&', '&')
732             .replace('<', '<')
733             .replace('>', '>');
734     }

768      * <p>Copies generated formatted HTML for an anchor tag to the clipboard.</p>
769      * @param {Tab} tab The tab whose information is to be used to generate the
770      * anchor tag.
771      * @requires jQuery
772      */
773     copyAnchor: function (tab) {
774         var data = {href: tab.url};

782             }
783             if (utils.get('settingTitleAttr')) {
784                 data.title = helper.addSlashes(data.text);
785             }
786         }
787         // TODO: Implement option to set targets as '_blank'
788         data.text = helper.replaceEntities(data.text);
789         clipboard.copy(helper.createAnchor(data));

795      * @param {Tab} tab The tab whose information is to be used to generate the
796      * <code>url</code> tag.
797      */

801             url: tab.url

805         }
806         if (data.text && utils.get('settingIeTabTitle')) {
807             data.text = ietab.extractTitle(data.text);
808         }
809         clipboard.copy(helper.createBBCode(data));
810     },

818         var url = tab.url;

823              * TODO: Test this works as expected considering it will already
824              * contains an encoded URI segment.

826             url = helper.encode(url);
827         }

850         var url = tab.url;
851         if (utils.get('settingIeTabExtract')) {
852             url = helper.decode(ietab.extractUrl(url));
853         }
854         clipboard.copy(url);
855     },
856 
857     /**
858      * <p>Attempts to extract the title embedded within that used by the IE Tab
859      * extension.</p>
860      * <p>If no title prefix was detected the string provided is returned.</p>
861      * @param {String} str The string from which to extract the embedded title.
862      * @returns {String} The title extracted from the string or the string if no
863      * prefix was detected.
864      * @private
865      */
866     extractTitle: function (str) {
867         var idx = str.indexOf(ietab.titlePrefix);
868         if (idx !== -1) {
869             return str.substring(idx + ietab.titlePrefix.length);
870         }
871         return str;
872     },
873 
874     /**
875      * <p>Attempts to extract the URL embedded within that used by the IE Tab
876      * extension.</p>
877      * <p>The embedded URL is returned as-is and is not decoded prior to being
878      * returned. If no embedded URL was found the string provided is returned.
879      * </p>
880      * @param {String} str The string from which to extract the embedded URL.
881      * @returns {String} The URL extracted from the string or the string if no
882      * embedded URL was found.
883      * @private
884      */
885     extractUrl: function (str) {
886         var idx = str.indexOf(ietab.containerSegment);
887         if (idx !== -1) {
888             return str.substring(idx + ietab.containerSegment.length);
889         }
890         return str;
891     },
892 
893     /**
894      * <p>Determines whether or not the IE Tab extension is currently active on
895      * the tab provided.</p>
896      * @param {Tab} tab The tab to be tested.
897      * @returns {Boolean} <code>true</code> if the IE Tab extension is active;
898      * otherwise <code>false</code>.
899      */
900     isActive: function (tab) {
901         return clipboard.isSpecialPage(tab) &&
902                 tab.url.indexOf(ietab.extensionId) !== -1;
903     }
