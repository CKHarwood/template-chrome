1 // Injects listener for extension keyboard shortcuts in to page context
  2 (function () {
  3     // Only adds listener if previous injection isn't detected
  4     if (document.body.hasAttribute('url_copy_injected')) {
  5         return;
  6     }
  7     document.body.setAttribute('url_copy_injected', true);
  8     /*
  9      * TODO: Confirm 'keydown' is best choice (e.g. 'keyup').
 10      * Verify keyCodes are still correct if 'keyup' is used instead.
 11      */
 12     document.body.addEventListener('keydown', function (event) {
 13         var isMac = navigator.userAgent.toLowerCase().indexOf('mac') !== -1;
 14         if ((event.ctrlKey && event.altKey && !isMac) ||
 15             (event.metaKey && event.altKey && isMac)) {
 16             var data = {'shortcut': true};
 17             /*
 18              * Determines the requested feature before serving it to extension's
 19              * request handler.
 20              */
 21             switch (event.keyCode) {
 22                 case 85: // 'U'
 23                     data.feature = 'copy_url';
 24                     break;
 25                 case 83: // 'S'
 26                     data.feature = 'copy_short';
 27                     break;
 28                 case 65: // 'A'
 29                     data.feature = 'copy_anchor';
 30                     break;
 31                 case 66: // 'B'
 32                     data.feature = 'copy_bbcode';
 33                     break;
 34                 case 69: // 'E'
 35                     data.feature = 'copy_encoded';
 36                     break;
 37             }
 38             if (data.feature) {
 39                  chrome.extension.sendRequest(data);
 40             }
 41         }
 42     }, false);
 43 }());
