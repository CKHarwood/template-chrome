101     /**
102      * <p>Updates the feature section of the options page with the current
103      * settings.</p>
104      * @requires jQuery
105      * @private
106      */
107     loadFeatures: function () {
108         var bg = chrome.extension.getBackgroundPage();
109         // Ensures clean slate
110         $('#settingFeatures option').remove();
111         // Creates and inserts options representing features
112         for (var i = 0; i < bg.clipboard.features.length; i++) {
113             var feature = bg.clipboard.features[i];
114             var opt = $('<option/>', {
115                 'text': chrome.i18n.getMessage(feature.name),
116                 'value': feature.name
117             }).appendTo('#settingFeatures');
118             opt.data('enabled', String(feature.isEnabled()));
119         }
120         /*
121          * Whenever the selected option changes we want all the controls to
122          * represent the current selection (where possible).
123          */
124         $('#settingFeatures').change(function (event) {
125             var opt = $(this).find('option:selected');
126             // Disables all the controls as no option is selected
127             if (opt.length === 0) {
128                 $('#moveDownButton, #moveUpButton, #settingFeatureEnabled')
129                         .attr('disabled', 'disabled');
130             } else {
131                 // Disables 'up' control as option is at top of the list
132                 if (opt.is(':first-child')) {
133                     $('#moveUpButton').attr('disabled', 'disabled');
134                 } else {
135                     $('#moveUpButton').removeAttr('disabled');
136                 }
137                 // Disables 'down' control as option is at bottom of the list
138                 if (opt.is(':last-child')) {
139                     $('#moveDownButton').attr('disabled', 'disabled');
140                 } else {
141                     $('#moveDownButton').removeAttr('disabled');
142                 }
143                 // Updates checkbox control to reflect if the feature's enabled
144                 if (opt.data('enabled') === 'true') {
145                     $('#settingFeatureEnabled').attr('checked', 'checked');
146                 } else {
147                     $('#settingFeatureEnabled').removeAttr('checked');
148                 }
149                 // Disables checkbox control if option is for 'copy_url' feature
150                 if (bg.feature.url.name === opt.val()) {
151                     $('#settingFeatureEnabled').attr('disabled', 'disabled');
152                 } else {
153                     $('#settingFeatureEnabled').removeAttr('disabled');
154                 }
155             }
156             /*
157              * Ensures correct arrows display in 'up'/'down' controls depending
158              * on whether or not the controls are disabled.
159              */
160             // TODO: Is this required on every change or just once?
161             $('#moveDownButton:not([disabled]) img').attr('src',
162                     '../images/move_down.gif');
163             $('#moveUpButton:not([disabled]) img').attr('src',
164                     '../images/move_up.gif');
165             $('#moveDownButton[disabled] img').attr('src',
166                     '../images/move_down_disabled.gif');
167             $('#moveUpButton[disabled] img').attr('src',
168                     '../images/move_up_disabled.gif');
169         }).change();
170         // Moves the selected option down one when the 'down' control is clicked
171         $('#moveDownButton').click(function (event) {
172             var opt = $('#settingFeatures option:selected');
173             opt.insertAfter(opt.next());
174             $('#settingFeatures').change();
175         });
176         // Moves the selected option up one when the 'up' control is clicked
177         $('#moveUpButton').click(function (event) {
178             var opt = $('#settingFeatures option:selected');
179             opt.insertBefore(opt.prev());
180             $('#settingFeatures').change();
181         });
182         /*
183          * Updates the 'enabled' data bound to the selected option when the
184          * checkbox control is clicked.
185          */
186         $('#settingFeatureEnabled').click(function (event) {
187             $('#settingFeatures option:selected').data('enabled',
188                     String($(this).is(':checked')));
189         });
190     },

192     /**
193      * <p>Updates the notification section of the options page with the current
194      * settings.</p>
195      * @requires jQuery
196      * @private
197      */
198     loadNotifications: function () {
199         if (utils.get('settingNotification')) {
200             $('#settingNotification').attr('checked', 'checked');
201         } else {
202             $('#settingNotification').removeAttr('checked');
203         }
204         var timeInSecs = 0;
205         if (utils.get('settingNotificationTimer') > timeInSecs) {
206             timeInSecs = utils.get('settingNotificationTimer') / 1000;
207         }
208         // Ensures clean slate
209         $('#settingNotificationTimer option').remove();
210         // Creates and inserts options for available auto-hide time in seconds
211         for (var i = 0; i <= 30; i++) {
212             var opt = $('<option/>', {
213                 'text': i,
214                 'value': i
215             });
216             if (i === timeInSecs) {
217                 opt.attr('selected', 'selected');
218             }
219             opt.appendTo('#settingNotificationTimer');
220         }
221     },

223     /**
224      * <p>Updates the URL shorteners section of the options page with the
225      * current settings.</p>
226      * @requires jQuery
227      * @private
228      */
229     loadUrlShorteners: function () {
230         $('input[name="settingEnabledUrlShortener"]').each(function () {
231             var radio = $(this);
232             if (utils.get(radio.attr('id'))) {
233                 radio.attr('checked', 'checked');
234             }
235         });
236         $('#bitlyXApiKey').val(utils.get('bitlyXApiKey'));
237         $('#bitlyXLogin').val(utils.get('bitlyXLogin'));
238     },

240     /**
241      * <p>Updates the settings with the values from the options page.</p>
242      * @requires jQuery
243      */
244     save: function () {
245         utils.set('settingShortcut', $('#settingShortcut').is(':checked'));
246         utils.set('settingTargetAttr', $('#settingTargetAttr').is(':checked'));
247         utils.set('settingTitleAttr', $('#settingTitleAttr').is(':checked'));
248         utils.set('settingIeTabExtract',
249                 !$('#settingIeTabExtract').is(':checked'));
250         utils.set('settingIeTabTitle', $('#settingIeTabTitle').is(':checked'));
251         options.saveFeatures();
252         options.saveNotifications();
253         options.saveUrlShorteners();
254     },
255 
256     /**
257      * <p>Updates the settings with the values from the options page and closes
258      * the current tab.</p>
259      * @param {Event} [event] The event triggered.
260      * @event
261      * @requies jQuery
262      * @private
263      */
264     saveAndClose: function (event) {
265         options.save();
266         chrome.tabs.getSelected(null, function (tab) {
267             chrome.tabs.remove(tab.id);
268         });
269     },

271     /**
272      * <p>Updates the settings with the values from the feature section of the
273      * options page.</p>
274      * @requires jQuery
275      * @private
276      */
277     saveFeatures: function () {
278         var bg = chrome.extension.getBackgroundPage();
279         /*
280          * Updates each individual feature settings based on their corresponding
281          * options.
282          */
283         $('#settingFeatures option').each(function (index) {
284             var opt = $(this);
285             switch (opt.val()) {
286                 case bg.feature.anchor.name:
287                     utils.set('copyAnchorEnabled',
288                             opt.data('enabled') === 'true');
289                     utils.set('copyAnchorOrder', index);
290                     break;
291                 case bg.feature.bbcode.name:
292                     utils.set('copyBBCodeEnabled',
293                             opt.data('enabled') === 'true');
294                     utils.set('copyBBCodeOrder', index);
295                     break;
296                 case bg.feature.encoded.name:
297                     utils.set('copyEncodedEnabled',
298                             opt.data('enabled') === 'true');
299                     utils.set('copyEncodedOrder', index);
300                     break;
301                 case bg.feature.short.name:
302                     utils.set('copyShortEnabled',
303                             opt.data('enabled') === 'true');
304                     utils.set('copyShortOrder', index);
305                     break;
306                 case bg.feature.url.name:
307                     utils.set('copyUrlEnabled', opt.data('enabled') === 'true');
308                     utils.set('copyUrlOrder', index);
309                     break;
310             }
311         });
312         // Ensures features data reflects the updated settings
313         bg.clipboard.updateFeatures();
314     },

316     /**
317      * <p>Updates the settings with the values from the notification section of
318      * the options page.</p>
319      * @requires jQuery
320      * @private
321      */
322     saveNotifications: function () {
323         utils.set('settingNotification',
324                 $('#settingNotification').is(':checked'));
325         var timeInSecs = $('#settingNotificationTimer').val();
326         timeInSecs = (timeInSecs) ? parseInt(timeInSecs, 10) * 1000 : 0;
327         utils.set('settingNotificationTimer', timeInSecs);
328     },

340         });
