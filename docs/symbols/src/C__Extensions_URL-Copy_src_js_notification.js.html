1 /**
  2  * <p>Responsible for the notification page.</p>
  3  * @author <a href="http://github.com/neocotic">Alasdair Mercer</a>
  4  * @since 0.0.2.1
  5  * @namespace
  6  */
  7 var notification = {
  8 
  9     /**
 10      * <p>Initializes the notification page.</p>
 11      * <p>This involves inserting and configuring the UI elements as well as the
 12      * insertion of localized Strings based on the result of the copy request.
 13      * </p>
 14      * @requires jQuery
 15      * @see clipboard.showNotification
 16      */
 17     init: function () {
 18         var bg = chrome.extension.getBackgroundPage();
 19         var result = (bg.clipboard.status) ? 'copy_success' : 'copy_fail';
 20         var sub = chrome.i18n.getMessage(bg.clipboard.feature + '_notification');
 21         /*
 22          * Styles the tip and inserts relevant localized String depending on the
 23          * result of the copy request.
 24          */
 25         $('#tip').addClass(result).html(chrome.i18n.getMessage(result, sub));
 26         /*
 27          * Important: Resets the clipboard to avoid affecting copy requests.
 28          * If user disabled the notifications option this is still called in
 29          * clipboard.showNotification for safety.
 30          */
 31         bg.clipboard.reset();
 32         /*
 33          * Sets a timer to close the notification after if user enabled option;
 34          * otherwise stays open until closed manually by user.
 35          */
 36         if (utils.get('settingNotificationTimer') > 0) {
 37             window.setTimeout(function () {
 38                 window.close();
 39             }, utils.get('settingNotificationTimer'));
 40         }
 41     }
 42 
 43 };
