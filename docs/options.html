loadFeature = (feature) ->
  opt = $ '<option/>',
    text:  feature.title
    value: feature.name
  opt.data 'content',  feature.content
  opt.data 'enabled',  String feature.enabled
  opt.data 'image',    String feature.image
  opt.data 'readOnly', String feature.readOnly
  opt.data 'shortcut', feature.shortcut
  return opt
